rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ===== RESULTS COLLECTION =====
    // Stores evaluation results from Notebook 2
    match /results/{document=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
      allow delete: if false;  // Never delete evaluation results
    }
    
    // ===== TEAM CONFIGURATIONS =====
    // Stores shared dataset info and team settings
    match /team_configs/{document=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    
    // ===== REPORTS =====
    // Stores report metadata
    match /reports/{document=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
    
    // ===== DEFAULT =====
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
